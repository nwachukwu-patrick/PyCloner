# PyCloner

This is website cloning software created with Python


# Usage

#### Cloning a single page 

url = 'https://example.com/'

dir  =  'output directory'

clone = Cloner(url,dir)

clone.webpage()

clone.clone()

#### Cloning an entire website

url = 'https://example.com/'

dir  =  'output directory'

clone = Cloner(url,dir)

clone.website()

clone.clone()
